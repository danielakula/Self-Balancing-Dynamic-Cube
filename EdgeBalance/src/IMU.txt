#include <Arduino.h>
#include "Config.h"
#include "IMU.h"

SPIClass hwSpi(1);

// Instantiate the IMU with the shared hardware SPI bus pointer
IMU_Sensor imu(ISM_CS, &hwSpi);

void setup() {
    Serial.begin(115200);
    delay(2000); 
    
    Serial.println("\n\n=======================================");
    Serial.println("       IMU ISOLATION TEST MODE         ");  
    Serial.println("=======================================\n");

    // 1. Spin up the hardware SPI bus
    hwSpi.begin(MASTER_SCK, MASTER_MISO, MASTER_MOSI, -1);
    
    // 2. Initialize the IMU (Motors are completely ignored and remain OFF)
    if (!imu.init()) {
        Serial.println("IMU init failed! Check your wiring.");
        while (1) delay(10);
    }
    
    // 3. Calibrate (Keep the robot perfectly still on its side/face!)
    imu.calibrate();
    
    Serial.println("Calibration complete. Open Serial Plotter NOW!");
}

void loop() {
    static uint32_t lastPrintTime = 0;
    
    // 1. Run the filter math as fast as the CPU allows
    imu.update(); 
    
    // 2. Print at 50 Hz (every 20ms) for a buttery smooth Serial Plotter graph
    if (millis() - lastPrintTime >= 20) {
        lastPrintTime = millis();
        
        // The exact comma-separated format the Serial Plotter needs:
        Serial.printf("Angle:%.2f, PitchRate:%.2f\n", 
                      imu.getPitch()*RAD_TO_DEG, 
                      imu.getPitchRate()*RAD_TO_DEG);
    }
}
